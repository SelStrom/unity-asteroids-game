    6. **EcsGunSystem.cs** -- расширить: после `gun.ValueRW.CurrentShoots--;` вместо комментария "OnShooting callback -- Phase 5 Bridge Layer" добавить запись GunShootEvent в singleton buffer. Система должна стать ISystem **без** BurstCompile (per D-17 Phase 4 -- уже без Burst). Нужно получить singleton buffer через `SystemAPI.GetSingletonBuffer<GunShootEvent>()`. Для определения IsPlayer: итерировать с `WithEntityAccess()` и проверять `state.EntityManager.HasComponent<ShipTag>(entity)`. Полная новая реализация OnUpdate:
    ```csharp
    public void OnUpdate(ref SystemState state)
    {
        var deltaTime = SystemAPI.Time.DeltaTime;
        var gunEvents = SystemAPI.GetSingletonBuffer<GunShootEvent>();

        foreach (var (gun, entity) in SystemAPI.Query<RefRW<GunData>>().WithEntityAccess())
        {
            if (gun.ValueRO.CurrentShoots < gun.ValueRO.MaxShoots)
            {
                gun.ValueRW.ReloadRemaining -= deltaTime;
                if (gun.ValueRO.ReloadRemaining <= 0)
                {
                    gun.ValueRW.ReloadRemaining = gun.ValueRO.ReloadDurationSec;
                    gun.ValueRW.CurrentShoots = gun.ValueRO.MaxShoots;
                }
            }

            if (gun.ValueRO.Shooting && gun.ValueRO.CurrentShoots > 0)
            {
                gun.ValueRW.CurrentShoots--;
                gunEvents.Add(new GunShootEvent
                {
                    ShooterEntity = entity,
                    Position = gun.ValueRO.ShootPosition,
                    Direction = gun.ValueRO.Direction,
                    IsPlayer = state.EntityManager.HasComponent<ShipTag>(entity)
                });
            }

            gun.ValueRW.Shooting = false;
        }
    }
    ```
    Также добавить в OnCreate: `state.RequireForUpdate<GunShootEvent>();` (RequireForUpdate для singleton buffer entity).

    using Unity.Entities;

    namespace SelStrom.Asteroids.ECS
    {
        [UpdateAfter(typeof(EcsLifeTimeSystem))]
        public partial class EcsDeadByLifeTimeSystem : SystemBase
        {
            protected override void OnUpdate()
            {
                var ecb = new EntityCommandBuffer(Unity.Collections.Allocator.Temp);

                foreach (var (lifeTime, entity) in
                         SystemAPI.Query<RefRO<LifeTimeData>>()
                             .WithNone<DeadTag>()
                             .WithEntityAccess())
                {
                    if (lifeTime.ValueRO.TimeRemaining <= 0f)
                    {
                        ecb.AddComponent<DeadTag>(entity);
                    }
                }

                ecb.Playback(EntityManager);
                ecb.Dispose();
            }
        }
    }
