EcsRocketGuidanceSystem -- partial class : SystemBase (НЕ ISystem struct, т.к. нужны managed-вызовы EntityManager.Exists/HasComponent -- per STATE.md).

Атрибуты: `[UpdateAfter(typeof(EcsMoveSystem))]` -- позиции врагов уже обновлены в текущем кадре.

**OnUpdate:**
1. Итерация по `SystemAPI.Query<RefRW<MoveData>, RefRW<RocketTargetData>>().WithAll<RocketTag>().WithNone<DeadTag>().WithEntityAccess()`
2. Проверка валидности цели: если Target != Entity.Null, проверить `EntityManager.Exists(target) && !EntityManager.HasComponent<DeadTag>(target)`. Если невалидна -- сбросить Target = Entity.Null.
3. Если Target == Entity.Null -- вызвать FindClosestEnemy(rocketPosition).
4. Если Target != Entity.Null -- получить позицию цели через EntityManager.GetComponentData<MoveData>(target).Position, вычислить вектор к цели, повернуть Direction на максимум turnRate * deltaTime градусов.
5. Если Target == Entity.Null после поиска -- Direction не меняется (ракета летит прямо, per D-06).

**FindClosestEnemy(float2 rocketPosition):**
3 отдельных foreach по AsteroidTag, UfoBigTag, UfoTag (все с `.WithNone<DeadTag>()`). Для каждого вычислить math.distancesq. Вернуть Entity с минимальным distancesq или Entity.Null если врагов нет.

**RotateTowards(float2 current, float2 target, float turnRateDeg, float deltaTime):**
Статический метод. Cross-product подход:
1. maxAngle = math.radians(turnRateDeg * deltaTime)
2. cross = current.x * target.y - current.y * target.x (знак = направление поворота)
3. dot = math.dot(current, target); angle = math.acos(math.clamp(dot, -1f, 1f))
4. Если angle <= maxAngle -- вернуть target (достигли цели)
5. rotAngle = math.sign(cross) * maxAngle; вращение через cos/sin:
   result.x = current.x * cos - current.y * sin
   result.y = current.x * sin + current.y * cos
6. Вернуть math.normalizesafe(result)

Защита от normalizesafe(float2.zero): если toTarget == float2.zero (ракета и цель на одной точке) -- не менять Direction (Pitfall 2 из RESEARCH.md).

```csharp
protected override void OnUpdate()
{
    var deltaTime = SystemAPI.Time.DeltaTime;

    foreach (var (move, target, entity) in
             SystemAPI.Query<RefRW<MoveData>, RefRW<RocketTargetData>>()
                 .WithAll<RocketTag>()
                 .WithNone<DeadTag>()
                 .WithEntityAccess())
    {
        // 1. Проверить валидность цели
        if (target.ValueRO.Target != Entity.Null)
        {
            if (!EntityManager.Exists(target.ValueRO.Target) ||
                EntityManager.HasComponent<DeadTag>(target.ValueRO.Target))
            {
                target.ValueRW.Target = Entity.Null;
            }
        }

        // 2. Если цели нет -- найти ближайшего врага
        if (target.ValueRO.Target == Entity.Null)
        {
            target.ValueRW.Target = FindClosestEnemy(move.ValueRO.Position);
        }

        // 3. Если цель есть -- повернуть Direction к цели
        if (target.ValueRO.Target != Entity.Null)
        {
            var targetPos = EntityManager.GetComponentData<MoveData>(target.ValueRO.Target).Position;
            var toTarget = math.normalizesafe(targetPos - move.ValueRO.Position);
            if (!math.all(toTarget == float2.zero))
            {
                move.ValueRW.Direction = RotateTowards(
                    move.ValueRO.Direction,
                    toTarget,
                    target.ValueRO.TurnRateDegPerSec,
                    deltaTime);
            }
        }
        // Если цели нет -- Direction не меняется (per D-06)
    }
}
```

**FindClosestEnemy:**
```csharp
private Entity FindClosestEnemy(float2 rocketPosition)
{
    var closestEntity = Entity.Null;
    var closestDistSq = float.MaxValue;

    foreach (var (move, entity) in
             SystemAPI.Query<RefRO<MoveData>>()
                 .WithAll<AsteroidTag>()
                 .WithNone<DeadTag>()
                 .WithEntityAccess())
    {
        var distSq = math.distancesq(rocketPosition, move.ValueRO.Position);
        if (distSq < closestDistSq)
        {
            closestDistSq = distSq;
            closestEntity = entity;
        }
    }

    foreach (var (move, entity) in
             SystemAPI.Query<RefRO<MoveData>>()
                 .WithAll<UfoBigTag>()
                 .WithNone<DeadTag>()
                 .WithEntityAccess())
    {
        var distSq = math.distancesq(rocketPosition, move.ValueRO.Position);
        if (distSq < closestDistSq)
        {
            closestDistSq = distSq;
            closestEntity = entity;
        }
    }

    foreach (var (move, entity) in
             SystemAPI.Query<RefRO<MoveData>>()
                 .WithAll<UfoTag>()
                 .WithNone<DeadTag>()
                 .WithEntityAccess())
    {
        var distSq = math.distancesq(rocketPosition, move.ValueRO.Position);
        if (distSq < closestDistSq)
        {
            closestDistSq = distSq;
            closestEntity = entity;
        }
    }

    return closestEntity;
}
```

**RotateTowards:**
```csharp
private static float2 RotateTowards(float2 current, float2 target, float turnRateDeg, float deltaTime)
{
    var maxAngle = math.radians(turnRateDeg * deltaTime);

    var cross = current.x * target.y - current.y * target.x;
    var dot = math.dot(current, target);
    var angle = math.acos(math.clamp(dot, -1f, 1f));

    if (angle <= maxAngle)
    {
        return target;
    }

    var rotAngle = math.sign(cross) * maxAngle;
    var cos = math.cos(rotAngle);
    var sin = math.sin(rotAngle);
    return math.normalizesafe(new float2(
        current.x * cos - current.y * sin,
        current.x * sin + current.y * cos
    ));
}
```
