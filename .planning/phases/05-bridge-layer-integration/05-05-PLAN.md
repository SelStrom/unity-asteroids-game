    ```csharp
    /// <summary>
    /// Regression: In ECS mode, score was accumulated in ScoreData but never
    /// synced to Model.Score. GameScreen.ShowEndGame read Model.Score = 0.
    /// Fix: ObservableBridgeSystem syncs ScoreData -> Model.Score each frame.
    /// </summary>
    [Test]
    public void ScoreData_IsSynced_ToModelScore_ViaObservableBridge()
    {
        var model = new Model { GameArea = new Vector2(20f, 15f) };
        Assert.AreEqual(0, model.Score, "Score must start at 0");

        // Create ScoreData singleton with accumulated score
        var scoreEntity = m_Manager.CreateEntity();
        m_Manager.AddComponentData(scoreEntity, new ScoreData { Value = 350 });

        // Create ObservableBridgeSystem, set model
        var bridge = World.AddSystemManaged(new ObservableBridgeSystem());
        bridge.SetModel(model);

        // Need a ship entity for the system to not early-return
        // (system checks _hudData == null && _shipViewModel == null && _model == null)
        // Since _model is set, it won't early-return

        bridge.Update();

        Assert.AreEqual(350, model.Score,
            "ObservableBridgeSystem must sync ScoreData.Value to Model.Score");

        // Update score and verify re-sync
        m_Manager.SetComponentData(scoreEntity, new ScoreData { Value = 700 });
        bridge.Update();

        Assert.AreEqual(700, model.Score,
            "Model.Score must update when ScoreData changes");
    }
    ```

    NOTE: The test creates ObservableBridgeSystem as managed system in test World. Follow existing pattern from `AsteroidsEcsTestFixture`. If `World.AddSystemManaged` isn't available for `ObservableBridgeSystem` directly (it's partial SystemBase), create it via `World.GetOrCreateSystemManaged<ObservableBridgeSystem>()` and then call `bridge.SetModel(model)`.
