Система должна стать ISystem **без** BurstCompile (per D-17 Phase 4 -- уже без Burst). Нужно получить singleton buffer через `SystemAPI.GetSingletonBuffer<GunShootEvent>()`. Для определения IsPlayer: итерировать с `WithEntityAccess()` и проверять `state.EntityManager.HasComponent<ShipTag>(entity)`. Полная новая реализация OnUpdate:
    ```csharp
    public void OnUpdate(ref SystemState state)
    {
        var deltaTime = SystemAPI.Time.DeltaTime;
        var gunEvents = SystemAPI.GetSingletonBuffer<GunShootEvent>();

        foreach (var (gun, entity) in SystemAPI.Query<RefRW<GunData>>().WithEntityAccess())
        {
            if (gun.ValueRO.CurrentShoots < gun.ValueRO.MaxShoots)
            {
                gun.ValueRW.ReloadRemaining -= deltaTime;
                if (gun.ValueRO.ReloadRemaining <= 0)
                {
                    gun.ValueRW.ReloadRemaining = gun.ValueRO.ReloadDurationSec;
                    gun.ValueRW.CurrentShoots = gun.ValueRO.MaxShoots;
                }
            }

            if (gun.ValueRO.Shooting && gun.ValueRO.CurrentShoots > 0)
            {
                gun.ValueRW.CurrentShoots--;
                gunEvents.Add(new GunShootEvent
                {
                    ShooterEntity = entity,
                    Position = gun.ValueRO.ShootPosition,
                    Direction = gun.ValueRO.Direction,
                    IsPlayer = state.EntityManager.HasComponent<ShipTag>(entity)
                });
            }

            gun.ValueRW.Shooting = false;
        }
    }
    ```
    Также добавить в OnCreate: `state.RequireForUpdate<GunShootEvent>();` (RequireForUpdate для singleton buffer entity).

        [UpdateAfter(typeof(EcsLifeTimeSystem))]

    1. **GameObjectSyncSystem.cs** -- создать managed SystemBase (per D-04, D-05):
    ```csharp
    using Unity.Entities;
    using Unity.Mathematics;
    using UnityEngine;

    namespace SelStrom.Asteroids.ECS
    {
        [UpdateInGroup(typeof(PresentationSystemGroup))]
        public partial class GameObjectSyncSystem : SystemBase
        {
            protected override void OnUpdate()
            {
                // Entities с MoveData + RotateData + GameObjectRef (корабль, UFO)
                foreach (var (move, rotate, goRef) in
                         SystemAPI.Query<RefRO<MoveData>, RefRO<RotateData>, GameObjectRef>())
                {
                    var pos = move.ValueRO.Position;
                    goRef.Transform.position = new Vector3(pos.x, pos.y, goRef.Transform.position.z);

                    var rot = rotate.ValueRO.Rotation;
                    var angle = math.atan2(rot.y, rot.x) * Mathf.Rad2Deg;
                    goRef.Transform.rotation = Quaternion.Euler(0f, 0f, angle);
                }

                // Entities с MoveData + GameObjectRef, без RotateData (астероиды, пули)
                foreach (var (move, goRef) in
                         SystemAPI.Query<RefRO<MoveData>, GameObjectRef>()
                             .WithNone<RotateData>())
                {
                    var pos = move.ValueRO.Position;
                    goRef.Transform.position = new Vector3(pos.x, pos.y, goRef.Transform.position.z);
                }
            }
        }
    }
    ```
